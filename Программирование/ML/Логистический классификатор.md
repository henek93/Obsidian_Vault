```python
import numpy as np

class LogisticRegression:
    def __init__(self, learning_rate=0.01, max_iter=1000):
        self.learning_rate = learning_rate
        self.max_iter = max_iter
    
    def sigmoid(self, z):
        # –ò–∑–±–µ–≥–∞–µ–º –ø–µ—Ä–µ–ø–æ–ª–Ω–µ–Ω–∏—è
        return np.where(z >= 0, 
                       1 / (1 + np.exp(-z)), 
                       np.exp(z) / (1 + np.exp(z)))
    
    def fit(self, X, y):
        # –î–æ–±–∞–≤–ª—è–µ–º —Å—Ç–æ–ª–±–µ—Ü –µ–¥–∏–Ω–∏—Ü –¥–ª—è —Å–º–µ—â–µ–Ω–∏—è
        X_with_bias = np.c_[np.ones(X.shape[0]), X]
        
        # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –≤–µ—Å–æ–≤
        self.weights = np.zeros(X_with_bias.shape[1])
        
        for i in range(self.max_iter):
            # –ü—Ä—è–º–æ–π –ø—Ä–æ—Ö–æ–¥
            z = X_with_bias.dot(self.weights)
            predictions = self.sigmoid(z)
            
            # –í—ã—á–∏—Å–ª–µ–Ω–∏–µ –≥—Ä–∞–¥–∏–µ–Ω—Ç–∞
            gradient = X_with_bias.T.dot(predictions - y) / len(y)
            
            # –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –≤–µ—Å–æ–≤
            self.weights -= self.learning_rate * gradient
    
    def predict_proba(self, X):
        X_with_bias = np.c_[np.ones(X.shape[0]), X]
        return self.sigmoid(X_with_bias.dot(self.weights))
    
    def predict(self, X):
        return (self.predict_proba(X) >= 0.5).astype(int)
```
