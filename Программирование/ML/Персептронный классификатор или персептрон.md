### –†–µ–∞–ª–∏–∑–∞—Ü–∏—è —Å –Ω—É–ª—è

```python
import numpy as np

class SimplePerceptron:
    def __init__(self, learning_rate=0.1, max_iter=1000):
        self.learning_rate = learning_rate
        self.max_iter = max_iter
    
    def fit(self, X, y):
        # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –≤–µ—Å–æ–≤ –∏ —Å–º–µ—â–µ–Ω–∏—è
        self.weights = np.zeros(X.shape[1])
        self.bias = 0
        
        # –û–±—É—á–µ–Ω–∏–µ
        for epoch in range(self.max_iter):
            errors = 0
            for i in range(X.shape[0]):
                # –í—ã—á–∏—Å–ª–µ–Ω–∏–µ –ª–∏–Ω–µ–π–Ω–æ–π –∫–æ–º–±–∏–Ω–∞—Ü–∏–∏
                linear_output = np.dot(X[i], self.weights) + self.bias
                
                # –ü—Ä–∏–º–µ–Ω–µ–Ω–∏–µ —Å—Ç—É–ø–µ–Ω—á–∞—Ç–æ–π —Ñ—É–Ω–∫—Ü–∏–∏
                prediction = 1 if linear_output >= 0 else 0
                
                # –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –≤–µ—Å–æ–≤ –ø—Ä–∏ –æ—à–∏–±–∫–µ
                if prediction != y[i]:
                    errors += 1
                    if y[i] == 1:  # –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å 1, –ø—Ä–µ–¥—Å–∫–∞–∑–∞–ª–∏ 0
                        self.weights += self.learning_rate * X[i]
                        self.bias += self.learning_rate
                    else:  # –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å 0, –ø—Ä–µ–¥—Å–∫–∞–∑–∞–ª–∏ 1
                        self.weights -= self.learning_rate * X[i]
                        self.bias -= self.learning_rate
            
            # –û—Å—Ç–∞–Ω–æ–≤–∫–∞ –ø—Ä–∏ –æ—Ç—Å—É—Ç—Å—Ç–≤–∏–∏ –æ—à–∏–±–æ–∫
            if errors == 0:
                break
                
    def predict(self, X):
        linear_output = np.dot(X, self.weights) + self.bias
        return np.where(linear_output >= 0, 1, 0)
```
